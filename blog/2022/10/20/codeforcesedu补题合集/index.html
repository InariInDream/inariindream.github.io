void solve() {
  int n;
  cin >> n;
  vector<int> a(n);
  cin >> a;
  sort(a.begin(), a.end());

  int left = 0, right = (n + 1) / 2;

  auto check = [&](int k) {
    if (k == 0) return true;
    if (a[k - 1] != 1) return false;

    for (int i = k; i < 2 * k - 1; i++) {
      if (a[i] > i - k + 2) {
        return false;
      }
    }
    return true;
  };

  while (left <= right) {
    int k = (left + right) / 2;

    if (check(k)) left = k + 1;
    else right = k - 1;
  }
  cout << left - 1 << "\n";
}
